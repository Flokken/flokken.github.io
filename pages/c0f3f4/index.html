 class MatrixSum {
    private  int[][] sum;

    public MatrixSum(int[][] matrix) {
        int m = matrix.length, n = matrix[0].length;
        sum = new int[m + 1][n + 1];
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                sum[i + 1][j + 1] = sum[i + 1][j] + sum[i][j + 1] - sum[i][j] + matrix[i][j];
            }
        }
    }

    public int query1(int r1, int c1, int r2, int c2) {
        return sum[r2][c2] - sum[r2][c1] - sum[r1][c2] + sum[r1][c1];
    }

    public int query2(int r1, int c1, int r2, int c2) {
        return sum[r2 + 1][c2 + 1] - sum[r2 + 1][c1] - sum[r1][c2 + 1] + sum[r1][c1];
    }
    public void printSum() {
      for (int i = 0; i < sum.length; i++) {
          for (int j = 0; j < sum[0].length; j++) {
              System.out.print(sum[i][j] + " ");
          }
          System.out.println();
      }
  }
}

public class Solution {
    public static void main(String[] args) {
        int[][] matrix = {
            {1, 2, 3},
            {4, 5, 6},
            {7, 8, 9}
        };

        MatrixSum matrixSum = new MatrixSum(matrix);

        int result1 = matrixSum.query1(1, 1, 2, 2);
        int result2 = matrixSum.query2(1, 1, 2, 2);
        System.out.println("Result 1: " + result1);
        System.out.println("Result 2: " + result2);
        matrixSum.printSum();

    }
}

# 辨析两种定义方法

上面的状态定义中，由于数组从0开始，到就是个数。所以定义表示左上角为,右下角为的元素和。

这样定义可以无需单独处理第一行/第一列的元素和，只是定义的数组会增大1即

这样定义是一种很好的技巧

下面是我自己想的，定义为表示左上角为,右下角为的元素和，这样需要单独处理第一行/第一列

因为这样定义不能从0开始遍历，会导致负数下标，所以定义为的确更好 本质区别是什么?

Answer: 实际上这两种定义是等效的替换

表示的是当前状态，而我们这里由于状态时 从前往后转移，所以就要考虑，由于，会导致负数下标，所以要单独处理第一行/第一列。不过数组边界无需多加1即

而是在时去考虑这个未来状态，但是我们这里是从转移过来，所以不存在负数下标的问题，也就无需单独处理第一行/第一列。定义的数组会增大1即

class Solution {
public int countSubmatrices(int[][] grid, int k) {
    int n=grid.length;
    int m=grid[0].length;
    int [][] f = new int[n][m];
    ans=1;
    f[0][0]=grid[0][0];
    for(int i=1;i<n;i++){
        f[i][0]=f[i-1][0]+grid[i][0];
        }
    }
    for(int i=1;i<m;i++){
        f[0][i]=f[0][i-1]+grid[0][i];

    }
    for(int i=1;i<n;i++){
        for(int j=1;j<m;j++){
            f[i][j]=f[i-1][j]+f[i][j-1]-f[i-1][j-1]+grid[i][j];
        }
    }
}
}

class gridSum {
    private final int[][] sum;

    public gridSum(int[][] grid) {
        int m = grid.length, n = grid[0].length;
        sum = new int[m + 1][n + 1];//注意这里我们
        // 注意：如果 grid[i][j] 范围很大，需要使用 long
        for (int i = 0; i < m; i++) {
            for (int j = 0; j < n; j++) {
                sum[i + 1][j + 1] = sum[i + 1][j] + sum[i][j + 1] - sum[i][j] + grid[i][j];
            }
        }
    }

    // 返回左上角在 (r1,c1) 右下角在 (r2-1,c2-1) 的子矩阵元素和（类似前缀和的左闭右开）
    public int query(int r1, int c1, int r2, int c2) {
        return sum[r2][c2] - sum[r2][c1] - sum[r1][c2] + sum[r1][c1];
    }

    
    
    // 如果你不习惯左闭右开，也可以这样写
    // 返回左上角在 (r1,c1) 右下角在 (r2,c2) 的子矩阵元素和
    public int query2(int r1, int c1, int r2, int c2) {
        return sum[r2 + 1][c2 + 1] - sum[r2 + 1][c1] - sum[r1][c2 + 1] + sum[r1][c1];
    }
}

# 100237. 元素和小于等于 k 的子矩阵的数目

100237. 元素和小于等于 k 的子矩阵的数目 (opens new window)

class Solution {
    public int countSubmatrices(int[][] grid, int k) {
        int n=grid.length;
        int m=grid[0].length;
        int [][]sum = new int [n+1][m+1];
        if(grid[0][0]>k){
            return 0;
        }
        int ans=0;
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                sum[i+1][j+1]=sum[i+1][j]+sum[i][j+1]-sum[i][j]+grid[i][j];
                if(sum[i+1][j+1]<=k){
                    ans++;
                }
            }
        }
        return ans;
    }
}

# 1277. 统计全为 1 的正方形子矩阵

1277. 统计全为 1 的正方形子矩阵 (opens new window)

首先求出前缀和，然后枚举所有的正方形矩阵。

难点在于枚举**正方形子矩阵，**具体思路有两种。

正方形子矩阵的特点就是他的边长是一样的，因此可以在最外层枚举其边长

  • 先确定正方形子矩阵长度，然后枚举子矩阵的右下角的坐标(左闭右开)，即(r2,c2)
    • r1=r2-k,c1=c2-k
 //枚举所有子矩阵
for(int k=1;k<=Math.min(n,m);k++){
    for(int i=k;i<n+1;i++){
        for(int j=k;j<m+1;j++){
            int r1=i-k,c1=j-k;
            int r2=i,c2=j;
            int res = query(r1,c1,r2,c2);
            if(res==k*k){
                ans++;
            }
        }
    }
}

# 221. 最大正方形

221. 最大正方形 (opens new window)

class Solution {
    private int [][]sum;
    public int maximalSquare(char[][] matrix) {
        int n=matrix.length;
        int m=matrix[0].length;
        sum=new int[n+1][m+1];
        for(int i=0;i<n;i++){
            for(int j=0;j<m;j++){
                sum[i+1][j+1]=sum[i+1][j]+sum[i][j+1]-sum[i][j]+matrix[i][j]-'0';
            }
        }
        int ans=0;
        for(int k=1;k<=Math.min(n,m);k++){
            for(int i=k;i<n+1;i++){
                for(int j=k;j<m+1;j++){
                    int r1=i-k,c1=j-k;
                    int res = query(r1,c1,i,j);
                    if(res==k*k){//都是1
                        ans=Math.max(ans,res);
                    }
                }
            }
        }
        return ans;

    }
    public int query(int r1,int c1,int r2,int c2){
        return sum[r2][c2]-sum[r1][c2]-sum[r2][c1]+sum[r1][c1];
    }
}
