    if (root != null) {
        queue.add(root);

        int n = queue.size();
        List<Integer> level = new ArrayList<>();
        for (int i = 0; i < n; i++) {

                queue.add(node.left);
            }

                queue.add(node.right);
            }
        }
        res.add(level);
    }
    return res;
}
