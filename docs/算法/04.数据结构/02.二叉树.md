    if (root != null) {
        queue.add(root);

        int n = queue.size();
        List<Integer> level = new ArrayList<>();
        for (int i = 0; i < n; i++) {

                queue.add(node.left);
            }

                queue.add(node.right);
            }
        }
        res.add(level);
    }
    return res;
}

            path.append(s2);
            return;
        }
        s2.append('L');
        dfs2(root.left,t);
        s2.deleteCharAt(s2.length()-1);

        s2.append('R');
        dfs2(root.right,t);
        s2.deleteCharAt(s2.length()-1);
    }
}
